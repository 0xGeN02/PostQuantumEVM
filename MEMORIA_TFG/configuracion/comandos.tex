\newcommand{\paginaenblancofrontmatter}{%
  \ifodd\value{page}
    \newpage
    \null
    \thispagestyle{frontblank}
    \newpage
  \else
    \clearpage
  \fi
}

\newcommand{\seccionfrontmatter}[2]{%
  \clearpage
  \pagestyle{frontmatter}
  \chapter*{#1}
  \thispagestyle{frontmatter}
  \addcontentsline{toc}{chapter}{#1}
  #2
  \paginaenblancofrontmatter
}
